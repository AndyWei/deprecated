### This file contains all curl commands used for manual test

## signin (GET a token)
curl -i -u jack@gmail.com:password "http://localhost:8000/v1/signin"




## orders

# GET all of the user's orders:
curl -i -H "Authorization: Bearer TOKEN" "http://localhost:8000/v1/orders/from_me"

# GET orders nearby
curl -i "http://localhost:8000/v1/orders/nearby?startPointLon=-122.4376&startPointLat=37.7577"

curl -i "http://localhost:8000/v1/orders/nearby?startPointLon=-122.4176&startPointLat=37.7577&categories=1"
curl -i "http://localhost:8000/v1/orders/nearby?startPointLon=-122.4176&startPointLat=37.7577&categories=1&categories=2&categories=5"

# create an order
curl -i -H "Authorization: Bearer MI4G7T4QXpSuxDuuonpO" --data "price=1.1&currency=usd&country=us&category=1&note='cleaning all'&title='cleaning for 4 rooms'&startTime=450700000&startAddress='2290 good ave'&startPointLon=-3.456&startPointLat=75.84" "http://localhost:8000/v1/orders"

# revoke an order
curl -i -X POST -H "Authorization: Bearer TOKEN" "http://localhost:8000/v1/orders/revoke/1"

# update an order
curl -i -X POST -H "Authorization: Bearer 6SjrTCgJPqFqOwdblR4Y" "http://localhost:8000/v1/orders/3"
curl -i -X POST -H "Authorization: Bearer 6SjrTCgJPqFqOwdblR4Y" --data "price=1.1&category=1" "http://localhost:8000/v1/orders/3"
curl -i -X POST -H "Authorization: Bearer 6SjrTCgJPqFqOwdblR4Y" --data "price=1.1&category=1&note='what ever:)'" "http://localhost:8000/v1/orders/3"



## bids

# POST an bid
curl -i -H "Authorization: Bearer TOKEN" --data "order_id=3&price=1.1&description='CHOOSE ME'" "http://localhost:8000/v1/bids"

# revoke an bid
curl -i -X POST -H "Authorization: Bearer TOKEN" "http://localhost:8000/v1/bid/revoke/1"

# accept a bid
curl -i -X POST -H "Authorization: Bearer TOKEN" "http://localhost:8000/v1/bids/accept/4"
